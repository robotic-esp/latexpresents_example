    \begin{frame}{Lists \& Colours}
        \threecolumneven%
        {%
            \begin{itemize}
                \item Regular text
                \item \emph{Emphasized text}
                \item Alerted text:
                \begin{itemize}
                    \item \alertBlue{A blue alert}%
                    \item \alertGreen{A green alert}%
                    \item \alertYellow{A yellow alert}%
                    \item \alertOrange{An orange alert}%
                    \item \alertRed{A red alert}%
                    \begin{itemize}
                        \item \alertRed{\LaTeX{} and $\zeta$}
                    \end{itemize}
                \end{itemize}
                \item Regular text with math: $42$%
            \end{itemize}
        }%
        {%
            \begin{enumerate}
                \item Regular text
                \item \emph{Emphasized text}
                \item Alerted text:
                \begin{enumerate}
                    \item \alertBlue{A blue alert}%
                    \item \alertGreen{A green alert}%
                    \item \alertYellow{A yellow alert}%
                    \item \alertOrange{An orange alert}%
                    \item \alertRed{A red alert}%
                    \begin{enumerate}
                        \item \alertRed{\LaTeX{} and $\zeta$}
                    \end{enumerate}
                \end{enumerate}
                \item Regular text with math: $42$%
            \end{enumerate}
        }%
        {%
            \begin{description}[Emphasis]
                \item [Regular] Regular text
                \item [Emphasis] \emph{Emphasized text}
                \item [Alert] \alertBlue{Alerted text}
            \end{description}
        }%
    \end{frame}

                \item[\texttt{Huge}] \Huge asdf
